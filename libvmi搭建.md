# 指令：
	./autogen.sh
	./configure
	make
	sudo make install

# 依赖库：
## 1.安装libtool,autoconf,automake
	sudo apt-get install libtool
	sudo apt-get install autoconf
	sudo apt-get install automake
## 2.安装glib-2.0
	sudo apt-get install libglib2.0-dev
## 3.安装lex,yacc
	sudo apt-get install flex bison
